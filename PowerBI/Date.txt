DATE =
VAR MINYEAR = YEAR ( MIN ( SALES[ORDER DATE] ) )
VAR MAXYEAR = YEAR ( MAX ( SALES[ORDER DATE] ) )
RETURN
ADDCOLUMNS (
    FILTER (
        CALENDARAUTO( ),
        AND ( YEAR ( [DATE] ) >= MINYEAR, YEAR ( [DATE] ) <= MAXYEAR )
    ),
    "CALENDAR YEAR", "CY " & YEAR ( [DATE] ),
    "MONTH NAME", FORMAT ( [DATE], "MMMM" ),
    "MONTH NUMBER", MONTH ( [DATE] )
)
